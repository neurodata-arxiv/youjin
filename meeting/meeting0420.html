# univariate distance matrix
uni.dist <- function(vec){
  n <- length(vec)
  mat <- matrix(0, n, n)
  for(i in 1:(n-1)){
    for(j in (i+1):n){
      mat[i,j] <- sqrt((vec[i] - vec[j])^2)
      mat[j,i] <- mat[i,j]
    }
  }
  return(mat)
}
